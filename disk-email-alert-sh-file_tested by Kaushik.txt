#!/bin/sh
df -Ph | grep  '/dev/nvme0n1p1' | awk '{ print $5,$1 }' | while read output;
do
  echo $output
  used=$(echo $output | awk '{print $1}' | sed s/%//g)
  partition=$(echo $output | awk '{print $2}')
  if [ $used -ge 60 ]; then
  echo "The partition \"$partition\" on $(hostname) has used $used% at $(date)" | mail -s "Disk Space Alert: $used% Used On Diashi-Server" -a "From: 210kaushik@gmail.com" kush210@gmail.com
  fi
done
